function qrKodaIdEkle(qrCanvasElement, kartID) { // 1. Orijinal QR boyutlarını al const qrWidth = qrCanvasElement.width; const qrHeight = qrCanvasElement.height; // Alt tarafa yazı için 35 piksellik bir alan ekliyoruz const extraHeight = 35; // 2. Yeni ve birleşik bir Canvas oluşturuyoruz const finalCanvas = document.createElement('canvas'); finalCanvas.width = qrWidth; finalCanvas.height = qrHeight + extraHeight; const ctx = finalCanvas.getContext('2d'); // Arka planı beyaz yap ctx.fillStyle = "#ffffff"; ctx.fillRect(0, 0, finalCanvas.width, finalCanvas.height); // QR Kodu çiz ctx.drawImage(qrCanvasElement, 0, 0); // Altına ID Metnini Yaz ctx.fillStyle = "#1e293b"; // Şık koyu gri/lacivert ctx.font = "bold 14px 'Segoe UI', Arial, sans-serif"; ctx.textAlign = "center"; ctx.fillText("Masa No: " + kartID, qrWidth / 2, qrHeight + 22); // Artık 'finalCanvas' elimizdeki ID'li yeni QR resmi! return finalCanvas.toDataURL("image/png"); }
